84  void Remap()
85  {
86  // ICW1 - initialize PIC1, PIC2, and tell them to receive ICW 2,3,4 next
87  out8(MASTER_COMMAND_PORT, ICW1_CONFIG); // init flag asks PIC1 to read ICW2,3,4
88  out8(SLAVE_COMMAND_PORT, ICW1_CONFIG); // init flag asks PIC2 to read ICW2,3,4
89 
90  // ICW2 - remap IRQ
91  out8(MASTER_DATA_PORT, ICW2_MASTER_OFFSET); // specify new interrupt number for master
92  out8(SLAVE_DATA_PORT, ICW2_SLAVE_OFFSET); // specify new interrupt number for slave
93 
94  // ICW3 - inform master/slave about each other
95  out8(MASTER_DATA_PORT, ICW3_MASTER_SLAVE_POS); // tells master position of slave in cascade (use IRQ2)
96  out8(SLAVE_DATA_PORT, ICW3_SLAVE_ID); // tells slave its IRQ number assigned on master
97 
98  // ICW4 - set PIC operation
99  out8(MASTER_DATA_PORT, ICW4_CPU); // set master PIC to x86 mode
100  out8(SLAVE_DATA_PORT, ICW4_CPU); // set slave PIC to x86 mode
101 
102  // disable interrupts (bit set indicates masking of interrupt)
103  out8(MASTER_DATA_PORT, 0xff);
104  out8(SLAVE_DATA_PORT, 0xff);
105  out8(MASTER_DATA_PORT, ~ICW3_MASTER_SLAVE_POS); // enable IRQ2 as cascade on IRQ1
106  }

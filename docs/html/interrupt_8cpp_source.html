59  void Remap()
60  {
61  // ICW1 - initialize PIC1, PIC2, and tell them to receive ICW 2,3,4 next
62  out8(MASTER_COMMAND_PORT, ICW1_CONFIG); // init flag asks PIC1 to read ICW2,3,4
63  out8(SLAVE_COMMAND_PORT, ICW1_CONFIG); // init flag asks PIC2 to read ICW2,3,4
64 
65  // ICW2 - remap IRQ
66  out8(MASTER_DATA_PORT, ICW2_MASTER_OFFSET); // specify new interrupt number for master
67  out8(SLAVE_DATA_PORT, ICW2_SLAVE_OFFSET); // specify new interrupt number for slave
68 
69  // ICW3 - inform master/slave about each other
70  out8(MASTER_DATA_PORT, ICW3_MASTER_SLAVE_POS); // tells master position of slave in cascade (use IRQ2)
71  out8(SLAVE_DATA_PORT, ICW3_SLAVE_ID); // tells slave its IRQ number assigned on master
72 
73  // ICW4 - set PIC operation
74  out8(MASTER_DATA_PORT, ICW4_CPU); // set master PIC to x86 mode
75  out8(SLAVE_DATA_PORT, ICW4_CPU); // set slave PIC to x86 mode
76 
77  // disable interrupts (bit set indicates masking of interrupt)
78  out8(MASTER_DATA_PORT, 0xff);
79  out8(SLAVE_DATA_PORT, 0xff);
80 
81  UnmaskInterrupt(MASTER_DATA_PORT, ICW3_SLAVE_ID); // enable IRQ2 as cascade on IRQ1
82  UnmaskInterrupt(MASTER_DATA_PORT, INTRP::IVT::TIMER_INTERRUPT - ICW2_MASTER_OFFSET); // enable IRQ2 as cascade on IRQ1
83  }

85  void UnmaskInterrupt (uint16_t Port, uint8_t Irq)
86  {
87  uint8_t Imr = in8(Port);
88 
89  if (Port == MASTER_DATA_PORT)
90  {
91  Imr &= ~(1 << Irq);
92  }
93  else if (Port == SLAVE_DATA_PORT)
94  {
95  Imr &= ~(1 << (Irq - 8));
96  }
97  else
98  {
99  kassert(false, "Only PIC data port should be used for interrupt");
100  }
101 
102  out8(Port, Imr);

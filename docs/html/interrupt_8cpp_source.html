20  // refer http://www.brokenthorn.com/Resources/OSDevPic.html
21  // ICW1
22  // to configure PIC
23  /*
24  * | bit | on (off) |
25  * | 0 | expect ICW 4 (not expect) |
26  * | 1 | cascade with slave (only 1 PIC) |
27  * | 2 | 8 byte interval (4 byte) |
28  * | 3 | edge triggered (level) |
29  * | 4 | initialize PIC (not initialize) |
30  * | 5-7 | zero for x86 |
31  */
32  const uint8_t ICW1_ICW4NEEDED = 0x1; // expect ICW 4
33  const uint8_t ICW1_INIT = 0x10; // initialize PIC
34  const uint8_t ICW1_CONFIG = ICW1_ICW4NEEDED | ICW1_INIT;
35 
36  // ICW2
37  // to map base IRQ address to PIC
38  /*
39  * | bit | description |
40  * | 0-7 | specifies start of interrupt number, must be byte aligned on x86, thus 0-3 not used |
41  */
42  const uint8_t ICW2_MASTER_OFFSET = 0x20; // IVT 0-31 are reserved for intel exception
43  const uint8_t ICW2_SLAVE_OFFSET = 0x28; // master handles IRQ0-7 (IVT 0x20), slave handles IRQ8-15 (IVT 0x28)
44 
45  // ICW3
46  // to specify relationship of master/slave to slave/master
47  /* Master
48  * | bit | description |
49  * | 0-7 | IRQ position connected to slave |
50  *
51  * Slave
52  * | bit | description |
53  * | 0-2 | IRQ number assigned by master, starts from 1 |
54  * | 3-7 | must be zero |
55  */
56  const uint8_t ICW3_MASTER_SLAVE_POS = 0x4; // tells master that slave is in 3rd position (IRQ2)
57  const uint8_t ICW3_SLAVE_ID = 0x2; // tells slave that it is assigned as IRQ2 on master
58 
59  // ICW4
60  // to specify PIC operation
61  /*
62  * | bit | on (off) |
63  * | 0 | MCS-80/86 mode (x86 mode) |
64  * | 1 | perform EOI on last interrupt acknowledge |
65  * | 2-4 | not used |
66  * | 5-7 | must be 0 |
67  */
68  const uint8_t ICW4_CPU = 0x1; // 8086/8088 mode

71  void Remap()
72  {
73  // In protected mode,
74  // Bios sets master PIC interrupt number to 0x8:0xF
75  // Bios sets slave PIC interrupt number to 0x70:0x77
76  // However master's 0x8:0xF clashes with Intel reserved exception from 0x0:0x1F
77  // Thus we remap PIC's interrupt number to 0x20:0x27 (master) and 0x28:0x2F
78  // While we only care about ICW2, resetting PIC requires specifying all ICW1-4
79 
80  // ICW1 - initialize PIC1, PIC2, and tell them to receive ICW 2,3,4 next
81  out8(MASTER_COMMAND_PORT, ICW1_CONFIG); // init flag asks PIC1 to read ICW2,3,4
82  out8(SLAVE_COMMAND_PORT, ICW1_CONFIG); // init flag asks PIC2 to read ICW2,3,4
83 
84  // ICW2 - remap IRQ
85  out8(MASTER_DATA_PORT, ICW2_MASTER_OFFSET); // specify new interrupt number for master
86  out8(SLAVE_DATA_PORT, ICW2_SLAVE_OFFSET); // specify new interrupt number for slave
87 
88  // ICW3 - inform master/slave about each other
89  out8(MASTER_DATA_PORT, ICW3_MASTER_SLAVE_POS); // tells master position of slave in cascade (use IRQ2)
90  out8(SLAVE_DATA_PORT, ICW3_SLAVE_ID); // tells slave its IRQ number assigned on master
91 
92  // ICW4 - set PIC operation
93  out8(MASTER_DATA_PORT, ICW4_CPU); // set master PIC to x86 mode
94  out8(SLAVE_DATA_PORT, ICW4_CPU); // set slave PIC to x86 mode
95 
96  // disable interrupts (bit set indicates masking of interrupt)
97  out8(MASTER_DATA_PORT, 0xff);
98  out8(SLAVE_DATA_PORT, 0xff);
99  out8(MASTER_DATA_PORT, ~ICW3_MASTER_SLAVE_POS); // enable IRQ2 as cascade on IRQ1
100  }

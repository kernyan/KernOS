74  void SetGlobalDescriptorEntry(GdtEntry GdtTable[], const Selector SelectorTable[])
75  {
76  for (size_t Idx = 0; Idx < GDT_ENTRIES; ++Idx) {
77 
78  const auto &Sel = SelectorTable[Idx];
79 
80  GdtTable[Sel.m_Seg].BaseLow = (Sel.m_Base & 0xFFFF);
81  GdtTable[Sel.m_Seg].BaseMid = (Sel.m_Base >> 16) & 0xFF;
82  GdtTable[Sel.m_Seg].BaseHigh = (Sel.m_Base >> 24) & 0xFF;
83 
84  GdtTable[Sel.m_Seg].LimitLow = (Sel.m_Limit & 0xFFFF);
85  GdtTable[Sel.m_Seg].LimitHighAndGranularity = ((Sel.m_Limit >> 16) & 0x0F);
86 
87  GdtTable[Sel.m_Seg].LimitHighAndGranularity |= (Sel.m_Granularity & 0xF0);
88  GdtTable[Sel.m_Seg].Access = Sel.m_Access;
89  }

107  void Install_gdt()
108  {
109  constexpr struct Selector SelTable[GDT_ENTRIES] =
110  {
111  { Null, SEG_OFFSET(Null), NULL_BASE, NULL_LIMIT, AR::NULL_ACCESS, NULL_GRANULARITY },
112  { K_CS, SEG_OFFSET(K_CS), KERN_CS_BASE, KERN_CS_LIMIT, AR::KERN_CS_ACCESS, KERN_CS_GRANULARITY },
113  { K_DS, SEG_OFFSET(K_DS), KERN_DS_BASE, KERN_DS_LIMIT, AR::KERN_DS_ACCESS, KERN_DS_GRANULARITY },
114  { U_CS, SEG_OFFSET(U_CS), USER_CS_BASE, USER_CS_LIMIT, AR::USER_CS_ACCESS, USER_CS_GRANULARITY },
115  { U_DS, SEG_OFFSET(U_DS), USER_DS_BASE, USER_DS_LIMIT, AR::USER_DS_ACCESS, USER_DS_GRANULARITY }
116  };
117 
118  SetGlobalDescriptorEntry(gdt_table, SelTable);
119 
120  Load_gdt(gdt_table, GDT_ENTRIES * 8 - 1);
121 
122  kassert (ReadCR0() & DWord<CR0::PE>(), "PE bit of cr0 must be set for protected mode\n");
123 
124  // while grub sets CS, DS, ES, FS, GS, and SS, the exact values are undefined, thus manually set them
125  // http://www.gnu.org/software/grub/manual/multiboot/multiboot.html#Machine-state
126 
127  // set DS, ES, FS, GS, and SS
128  asm volatile (
129  "mov %%ax, %%ds\n\t"
130  "mov %%ax, %%es\n\t"
131  "mov %%ax, %%fs\n\t"
132  "mov %%ax, %%gs\n\t"
133  "mov %%ax, %%ss\n\t"
134  : // no output
135  : "a"(SelTable[K_DS].m_Offset)
136  : "memory"
137  );
138 
139  // set CS
140  asm volatile (
141  "ljmpl %0, $CS_LABEL\n\t" // dummy long jump to force kernel code selector update
142  "CS_LABEL:\n\t"
143  : // no output
144  : "n"(SelTable[K_CS].m_Offset) // SelTable is constexpr, thus allowed to be used as immediate integer operand
145  );
146  }
